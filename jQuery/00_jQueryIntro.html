jQuery

        자바스크립트를 간편하게 사용할 수 있도록 하는
        오픈소스 기반 자바스크립트 라이브러리

        사용 방법 1- 홈페이지 다운로드 src="jquery-3.7.1.min.js"

        사용 방법 2- CDN 주소 참조 활용 (Contents Delivery Network)
        script 에 type="text/javascript" src="https://code.jquery.com/jquery-3.7.1.min.js"
        를 속성으로 넣어준다.
        


        [기본 문법]

        1. id 문법

            $("#id이름").click(function(){

                //내용 작성
            })
        })

        [선택자 문법]
        
        1. css 선택
        $("태그명, #아이디명, .클래스명").css("style속성",설정할값);

        2.toggle
        $("태그명, #아이디명, .클래스명").toggle("속도조절"); //속도는 자유롭게, 생략도가능

        3. slideToggle 선택 사용
            주로 이미지를 대상으로 사용
        
        4. hide() show() 선택 사용
            숨기기 보이기

        5. append() 태그나 값을 추가할 때 사용

        6. val()
            input textarea select 같은 폼안에 들어가는 태그의 값을 가져오거나
            설정하는데에 사용
            값을 가져올 때 $(태그명, #아이디값, .클래스명).val();
                ->$("input").val()
                ->$("#idName").val()
                ->$(".className").val()
            값을 넣을 때는 $(태그명, #아이디값, .클래스명).val("새로운 값");
